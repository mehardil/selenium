def data():
    ulofsouplist  = webd.find_elements(By.CLASS_NAME , "nav-dropdown__item")

        list01.append(ppp.text)
        listref.append(ppp.get_attribute('href'))
        print("step1-done")
        
        
####################################### get all url        
producturl = []
productname = []

    for j in listref:

            productname.append(ttt.text)
